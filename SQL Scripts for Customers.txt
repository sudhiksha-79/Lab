CREATE TABLE CUSTOMER_CLASSIFICATION (
    CSTCL_ID BIGINT PRIMARY KEY AUTO_INCREMENT,

    CSTCL_TYP VARCHAR(100) NOT NULL,
    CSTCL_TYP_VALUE VARCHAR(100) NOT NULL,

    CSTCL_EFCTV_DT DATE,

    CSTCL_CRUD_VALUE CHAR(1),

    CSTCL_USER_ID VARCHAR(100),
    CSTCL_WS_ID VARCHAR(100),
    CSTCL_PRGM_ID VARCHAR(100),

    CSTCL_HOST_TS TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CSTCL_LOCAL_TS TIMESTAMP,
    CSTCL_ACPT_TS TIMESTAMP,
    CSTCL_ACPT_TS_UTC_OFS TIMESTAMP,

    CSTCL_UUID VARCHAR(100)
);

CREATE TABLE CUSTOMER_DETAILS (
    CUST_NO BIGINT PRIMARY KEY AUTO_INCREMENT,

    CUSTOMER_TYPE_ID BIGINT,

    CUST_FULL_NAME VARCHAR(200),
    CUST_DOB DATE,
    CUST_STATUS VARCHAR(50),

    CUST_CONTACT_NO VARCHAR(20),
    CUST_MOBILE_NO VARCHAR(20),
    CUST_EMAIL VARCHAR(100),
    CUST_COUNTRY VARCHAR(100),

    CST_EFCTV_DT DATE,
    CST_CRUD_VALUE CHAR(1),

    CST_USER_ID VARCHAR(100),
    CST_WS_ID VARCHAR(100),
    CST_PRGM_ID VARCHAR(100),

    CST_HOST_TS TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CST_LOCAL_TS TIMESTAMP,
    CST_ACPT_TS TIMESTAMP,
    CST_ACPT_TS_UTC_OFS TIMESTAMP,

    CST_UUID VARCHAR(100),

    CONSTRAINT FK_CUSTOMER_TYPE
    FOREIGN KEY (CUSTOMER_TYPE_ID)
    REFERENCES CUSTOMER_CLASSIFICATION(CSTCL_ID)
);

CREATE TABLE CUSTOMER_IDENTIFICATION (
    CUST_IDNT_ID BIGINT PRIMARY KEY AUTO_INCREMENT,

    CUSTOMER_ID BIGINT NOT NULL,
    ID_TYPE_ID BIGINT NOT NULL,

    CUST_IDNT_ITEM VARCHAR(100),

    CST_EFCTV_DT DATE,
    CST_CRUD_VALUE CHAR(1),

    CST_USER_ID VARCHAR(100),
    CST_WS_ID VARCHAR(100),
    CST_PRGM_ID VARCHAR(100),

    CST_HOST_TS TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CST_LOCAL_TS TIMESTAMP,
    CST_ACPT_TS TIMESTAMP,
    CST_ACPT_TS_UTC_OFS TIMESTAMP,

    CST_UUID VARCHAR(100),

    UNIQUE (ID_TYPE_ID, CUST_IDNT_ITEM),

    CONSTRAINT FK_ID_CUSTOMER
        FOREIGN KEY (CUSTOMER_ID)
        REFERENCES CUSTOMER_DETAILS(CUST_NO),

    CONSTRAINT FK_ID_TYPE
        FOREIGN KEY (ID_TYPE_ID)
        REFERENCES CUSTOMER_CLASSIFICATION(CSTCL_ID)
);
